def evaluate_model(y_true, y_pred, model_name):
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    mae = mean_absolute_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    
    print(f"\nMetrics for {model_name}:")
    print(f"RMSE: {rmse:.4f}")
    print(f"MAE: {mae:.4f}")
    print(f"R2 Score: {r2:.4f}")
    
    return {'RMSE': rmse, 'MAE': mae, 'R2': r2}

def preprocess_data(data):
    """
    Apply all preprocessing steps from the reference code
    """
    "['X3', 'X6', 'X7', 'X8', 'X12', 'X14', 'X15', 'X17', 'X19', 'X20', 'X21', 'X22', 'X23', 'X24', 'X28', 'X29', 'X31', 'X32', 'X13_filled_na', 'X25_filled_na']"
    # Drop unnecessary columns as per reference
    columns_to_drop = ['X16', 'X26', 'X9', 'X10', 'X11', 'X30', 'X18', "X27"]
    data = data.drop([col for col in columns_to_drop if col in data.columns], axis=1)
    
    if "X1" in data.columns:
        data = data.drop(["X1"], axis = 1)
    
    
    # KNN Imputation for X13 and X25
    imputer = KNNImputer(n_neighbors=5)
    data['X13_filled_na'] = imputer.fit_transform(data[['X13']])
    data['X25_filled_na'] = imputer.fit_transform(data[['X25']])
    
    
    # Drop original columns
    data = data.drop(columns=["X13", "X25"], axis=1)
    # Handle categorical missing values
    categorical_cols = data.select_dtypes(include=['object']).columns
    for col in categorical_cols:
        data[col].fillna(data[col].mode()[0], inplace=True)
    
    # Label encode categorical columns
    le = LabelEncoder()
    for col in categorical_cols:
        data[col] = le.fit_transform(data[col])
    
    # Drop highly correlated features identified in reference
    features_to_drop = ['X2', 'X4', 'X5']  # Based on correlation analysis
    data = data.drop(columns=[col for col in features_to_drop if col in data.columns])
    
    # Scale numerical columns
    numerical_cols = [col for col in data.columns if col.endswith('_filled_na')]
    scaler = StandardScaler()
    data[numerical_cols] = scaler.fit_transform(data[numerical_cols])
    
    return data

def make_predictions(data_path, model_path, output_path):
    """
    Load data, preprocess it, make predictions, and save results
    """
    # Load the data
    data = pd.read_csv(data_path)
    
    # Preprocess the data
    processed_data = preprocess_data(data)
    
    # Load the model
    model = load_model(model_path)
    
    # Make predictions
    predictions = model.predict(processed_data)
    
    # Create output dataframe
    output_df = pd.DataFrame({
        'Predicted_Value': predictions
    })
    print(output_df.head(5))
    # Save predictions
    output_df.to_csv(output_path, index=False)
    print(f"Predictions saved to {output_path}")
    
    return predictions
